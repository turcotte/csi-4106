import { Inputs, Plot } from "@observablehq/plot";

// Alpha slider ranging from 0 to 0.999
viewof alpha = Inputs.range([0, 0.999], {step: 0.001, value: 0.995, label: "Alpha", width: 300})

// Constants
initial_temp = 100

// Iterations from 0 to 1000
iterations = Array.from({length: 1001}, (_, i) => i)

// Cooling strategies
exponential_cooling = (initial_temp, alpha, iteration) =>
  initial_temp * Math.pow(alpha, iteration)

linear_cooling = (initial_temp, alpha, iteration) =>
  Math.max(0, initial_temp - alpha * iteration)

logarithmic_cooling = (initial_temp, alpha, iteration) =>
  Math.min(initial_temp, initial_temp * alpha / (Math.log(1 + iteration)))

inverse_cooling = (initial_temp, alpha, iteration) =>
  initial_temp / (1 + alpha * iteration)

// Calculate temperatures and prepare data for plotting
function getCoolingData(alpha) {
  return iterations.flatMap(i => [
    {
      iteration: i,
      temp: exponential_cooling(initial_temp, alpha, i),
      strategy: "Exponential Cooling"
    },
    {
      iteration: i,
      temp: linear_cooling(initial_temp, alpha, i),
      strategy: "Linear Cooling"
    },
    {
      iteration: i,
      temp: logarithmic_cooling(initial_temp, alpha, i),
      strategy: "Logarithmic Cooling"
    },
    {
      iteration: i,
      temp: inverse_cooling(initial_temp, alpha, i),
      strategy: "Inverse Cooling"
    }
  ]);
}

// Make coolingData reactive to alpha changes
coolingData = getCoolingData(alpha)

// Plotting using Observable Plot
Plot.plot({
  marks: [
    Plot.line(coolingData, {x: "iteration", y: "temp", stroke: "strategy"})
  ],
  x: {
    label: "Iteration",
    grid: true
  },
  y: {
    label: "Temperature",
    grid: true
  },
  color: {
    legend: true
  },
  caption: `Cooling Strategies (α = ${alpha})`
})
